\section{1\_200\_300s}

\begin{tikzpicture}
  \begin{axis}[
    title={All non-timeout runs},
    xlabel={Formula size},
    ylabel={Runtime (s)},
    ymode=log,
    log basis y={10},
    grid=both,
    width=12cm,
    height=7cm,
    legend style={at={(1.05,1)}, anchor=north west},
    legend cell align=left,
    scatter/classes={
      zip={mark=*,        color=blue},
      tree={mark=triangle*, color=red}
    },
  ]

    \addplot[
      scatter,
      only marks,
      scatter src=explicit symbolic,
    ]
    table[
      col sep=comma,
      x=size,
      y=runtime,
      meta=cat,
    ] {1_200_300s_scattered.csv};

    \addlegendimage{only marks,mark=*,color=blue}
    \addlegendentry{zip}
    \addlegendimage{only marks,mark=triangle*,color=red}
    \addlegendentry{tree}

  \end{axis}
\end{tikzpicture}
